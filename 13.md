### 动态表格

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>
    <table border="1">
        <thead>
        <tr>
            <th>ID</th>
            <th>姓名</th>
            <th>年龄</th>
        </tr>
        </thead>
        <tbody id="body">

        </tbody>
    </table>


    <script type="text/javascript">
        //网络请求获取，写入页面

        var datalist = [
            {id:1,name:"郭智",age:19},
            {id:2,name:"郭智",age:19},
            {id:3,name:"郭智",age:19}
        ]
        for (var idx in datalist) {

            var info = datalist[idx]

            var tr = document.createElement("tr")
            for (var key in info) {
                var text = info[key];
                var td = document.createElement("td");
                td.innerText = text;

                tr.appendChild(td);
            }

            var bodytag = document.getElementById("body")
            bodytag.appendChild(tr)
        }
    </script>
</body>
</html>
```
